1) Removes Duplicates Emp's Based on First Name and LastName.

	max or min ( with or direct ) + delete query
	  
	row_number ( with or direct ) + delete query

	ANS 1:
			with unique_employees as 
			
				( select min(id) as id from emp group by first_name,last_name )
				
			DELETE FROM emp WHERE id NOT IN ( select id from unique_employees);
			
			or
				DELETE FROM emp WHERE id NOT IN (

				select min(id) from emp group by fn,ln)
				
			or 
				DELETE FROM emp WHERE id NOT IN (
					SELECT MAX(id) FROM emp GROUP BY first_name, last_name
				); 
	ANS 2:
			with ranked_emp as 
			
				( select id, row_number() over ( partition by f_n,l_n order by id) as rn from emp)
				
			delete from emp where id in ( select id from ranked_emp where rn>1);
			
			WITH ranked_emp AS (
			SELECT id,
				   ROW_NUMBER() OVER (
					   PARTITION BY f_n, l_n
					   ORDER BY id
				   ) rn
			FROM emp
			)
			DELETE FROM emp
			WHERE id IN (
				SELECT id
				FROM ranked_emp
				WHERE rn > 1
			);
		or 
				WITH cte AS (
					SELECT *,
						   ROW_NUMBER() OVER (
							   PARTITION BY first_name, last_name
							   ORDER BY id
						   ) rn
					FROM emp
				)
				DELETE FROM cte
				WHERE rn > 1;
	ANS 3:
			DELETE FROM emp
			WHERE id IN (
				SELECT id FROM (
					SELECT id,
						   ROW_NUMBER() OVER (
							   PARTITION BY first_name, last_name
							   ORDER BY id
						   ) AS rn
					FROM emp
				) t
				WHERE rn > 1
			);

	3.1  ✔ Compare with manager salary ✔ Department-wise comparisons
	
	✅ 1. Compare Employee Salary with Manager Salary
			⭐ Using SELF JOIN (Best Approach)
			
					SELECT e.empno,
					   e.ename,
					   e.salary AS emp_salary,
					   m.ename AS manager_name,
					   m.salary AS manager_salary
					FROM emp e
					JOIN emp m ON e.mgr = m.empno
					WHERE e.salary > m.salary;
		
	✅ 2. Department-wise Salary Comparisons
	⭐ A. Employees earning more than department average
	
			SELECT e.empno,
			   e.ename,
			   e.department_id,
			   e.salary
			FROM emp e
			WHERE e.salary > (
				SELECT AVG(salary)
				FROM emp
				WHERE department_id = e.department_id
			);
	⭐ B. Using WINDOW FUNCTION (Better Performance)
	
			SELECT empno,
			   ename,
			   department_id,
			   salary
			FROM (
				SELECT e.*,
					   AVG(salary) OVER (PARTITION BY department_id) AS dept_avg
				FROM emp e
			) t
			WHERE salary > dept_avg;
	
	⭐ C. Employees earning highest in each department
	
			SELECT *
			FROM (
				SELECT e.*,
					   DENSE_RANK() OVER (
						   PARTITION BY department_id
						   ORDER BY salary DESC
					   ) AS rnk
				FROM emp e
			) t
			WHERE rnk = 1;
		
	⭐ D. Employees earning more than any employee in another department
	
			SELECT *
			FROM emp e
			WHERE salary > ANY (
				SELECT salary
				FROM emp
				WHERE department_id = 10
			);

	✅ Using RANK() (Best for Interview 🔥)
	
		SELECT deptno, avg_salary
		FROM (
			SELECT deptno,
				   AVG(sal) AS avg_salary,
				   RANK() OVER (ORDER BY AVG(sal) DESC) rnk
			FROM emp
			WHERE sal > 6000
			GROUP BY deptno
		)
		WHERE rnk = 1;
		
	Explanation
		WHERE sal > 6000 → filter employees first
		GROUP BY deptno → calculate per department
		AVG(sal) → get average salary
		ORDER BY DESC → highest first
		FETCH FIRST 1 ROW → top department
		
	or
