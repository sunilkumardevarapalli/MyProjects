# 4. Max Repeating Item in string

  
String input = "programming";

Character c =
        input.chars()
             .mapToObj(ch -> (char) ch)
             .collect(Collectors.groupingBy(
                     Function.identity(),
                     Collectors.counting()
             ))
             .entrySet()
             .stream()
             .max(Map.Entry.comparingByValue())
             .map(Map.Entry::getKey)
             .orElse(null);

# 14. Balanced Parenthesis
 
String str = "{[()]}";

Stack<Character> stack = new Stack<>();

for(char c : str.toCharArray()) {

    if(c=='(' || c=='{' || c=='[')
        stack.push(c);

    else {
        if(stack.isEmpty()) {
            System.out.println("Not Balanced");
            return;
        }

        char ch = stack.pop();

        if((c==')' && ch!='(') ||
           (c=='}' && ch!='{') ||
           (c==']' && ch!='[')) {
            System.out.println("Not Balanced");
            return;
        }
    }
}

System.out.println(stack.isEmpty() ? "Balanced" : "Not Balanced");

# 38. Rotate Last Two Numbers
 
for(int i=0;i<2;i++) {

    int last = arr[arr.length-1];

    for(int j=arr.length-1;j>0;j--)
        arr[j] = arr[j-1];

    arr[0] = last;
}
