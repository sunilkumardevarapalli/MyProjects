2.1 Removes Duplicates Emp's Based on First Name and LastName.  min, max, row_number

		with unique_employees as 
					
			( select min(id) as id from emp group by first_name,last_name )
			
		DELETE FROM emp WHERE id NOT IN ( select id from unique_employees);
		
		or
			DELETE FROM emp WHERE id NOT IN (

			select min(id) from emp group by fn,ln)
			
		or 
			DELETE FROM emp WHERE id NOT IN (
				SELECT MAX(id) FROM emp GROUP BY first_name, last_name
			); 
			
		---
	
		WITH ranked_emp AS (
			SELECT id,
				   ROW_NUMBER() OVER (
					   PARTITION BY f_n, l_n
					   ORDER BY id
				   ) rn
			FROM emp
		)
		DELETE FROM emp
		WHERE id IN (
			SELECT id
			FROM ranked_emp
			WHERE rn > 1
		);
		---
		DELETE FROM emp
			WHERE id IN (
				SELECT id FROM (
					SELECT id,
						   ROW_NUMBER() OVER (
							   PARTITION BY first_name, last_name
							   ORDER BY id
						   ) AS rn
					FROM emp
				) t
				WHERE rn > 1
			);

2.5  Employees earning more than any employee in another department

		SELECT empno,
			   ename,
			   department_id,
			   salary
		FROM emp e1
		WHERE salary > ANY (
			SELECT salary
			FROM emp e2
			WHERE e2.department_id <> e1.department_id
		);
		
		
	   SELECT empno,
       ename,
       department_id,
       salary
		FROM emp e1
		WHERE salary > ALL (
			SELECT salary
			FROM emp e2
			WHERE e2.department_id <> e1.department_id
		);
		
		Difference between ANY and ALL

		Suppose the salaries in other departments are:

		3000, 5000, 7000

		For an employee earning 6000:

		6000 > ANY (3000, 5000, 7000) → ✅ True (6000 > 3000 and 5000)
		6000 > ALL (3000, 5000, 7000) → ❌ False (6000 is not greater than 7000)
		
2.6 select all department numbers which has more number of employee than department 10

	SELECT deptno
	FROM emp
	GROUP BY deptno
	HAVING COUNT(*) > (
		SELECT COUNT(*)
		FROM emp
		WHERE deptno = 10
	);
	
2.7*** select department names which has more number of employes working than department 10.

	select d.deptName,count(*) from emp e

		join dept d on 
		
		e.dept_no = d.dept_no 
		
		grouping by d.deptName
		
		having count(*) > select count(*) from emp 
		
		which deptNo = 10);
		
2.8 find the dept with the highest average salary

	SELECT dept,
       AVG(salary) AS avg_salary
	FROM emp
	GROUP BY dept  -- upd here got avg for all dept's
	
	ORDER BY avg_salary DESC
	LIMIT 1;
	
2.9 find the department with the hiest average salay and salary grater than 6000

	SELECT deptno, AVG(sal) AS avg_salary
	FROM emp
	WHERE sal > 6000
	GROUP BY deptno
	ORDER BY avg_salary DESC
	FETCH FIRST 1 ROW ONLY;
	
2.10 caliculate average salary of managers and non managers ?

	select 
		case when jobTitle like 'MANAGER' then 'MANAGER'
		
		else 'NON-MANAGER'
		
		end as role_group,
		
		avg (salary) as avg_salary from emp group by role_group;
